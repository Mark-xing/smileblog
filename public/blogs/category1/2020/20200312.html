# 函数传参数是按值还是引用?数据类型或者对象类型都一样吗？

# 答案是按值传参数

# 基本数据类型怎么复制？

var simpleData1 = 18;
var simpleData2 = simpleData1;
simpleData2 = 19;
console.log(simpleData1); //打印出18

# 引用数据类型怎么复制？

var arr1 = ["zyp", "zyp", "zyp"];
var arr2 = arr1;
arr2[0] = "zyp2";
console.log(arr2); //打印出 ["zyp2", "zyp", "zyp"]
console.log(arr1); //打印出 ["zyp2", "zyp", "zyp"]

引用数据类型复制，是复制的内存地址，arr1和arr2指的同一个地址

# 那函数里面的的传值是怎么复制的呢？

var arr1 = ["zyp", "zyp", "zyp"];
var arr2 = arr1;
arr2 = ["zyp1", "zyp1", "zyp1"];
arr2[0] = "zyp2";
console.log(arr2); //打印出 ["zyp2", "zyp1", "zyp1"]
console.log(arr1); //打印出 ["zyp", "zyp", "zyp"]

在这里我们如果在var arr2 = arr1后，赋值给arr2新的数组，会发现最后输出arr1没有被改变。理解了这个就好理解下面的例子。

举个例子：

let a = 1;
let obj = {};
function first(a1) {
    a1 = 2;
}
first(a);

function second(obj1) {
    obj1 = { name: "噢噢噢噢", age: 121 };
    obj1.age = 0;
}
second(obj);
console.log(a); //输出 1
console.log(obj); // 输出 {}

这里可以看到obj并没有被改变，就和上面的例子一样在var arr2 = arr1按值赋值，进行接 下来的操作，这里是先var obj1=obj按值赋值，如果是按照引用赋值的话，这里新开辟一个 obj1新的堆地址时,obj就会改变

接下来你再理解下面的例子就会明白了。

function setName(obj) {
    obj.name = "aaa";
    var obj = new Object(); // 如果是按引用传递的,此处传参进来obj应该被重新引用新的内存单元
    obj.name = "ccc";
    return obj;
}

var person = new Object();
person.name = "bbb";
var newPerson = setName(person);
console.log(person.name + " | " + newPerson.name); // aaa | ccc
